## Install

```bash
npm install --save mezrmouse
```

## Usage

```jsx
import React, { Component } from 'react'

import Measure from 'mezrmouse'

class Example extends Component {
  render () {
    return (
      <Measure distance={100} onCloseUp={(percent) => this.distance = percent} />
    )
  }
}
```
